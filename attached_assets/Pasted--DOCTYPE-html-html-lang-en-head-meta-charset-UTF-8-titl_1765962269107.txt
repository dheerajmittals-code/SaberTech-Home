  <script>
    function scrollToForm() {
      const el = document.getElementById("contact-form");
      if (el) el.scrollIntoView({ behavior: "smooth" });
    }

    function openWhatsApp() {
      // Replace with your business WhatsApp number in international format
      const phone = "919999999999"; // TODO: update this
      const text = encodeURIComponent("Hi, I’d like to discuss outsourcing / gig / operations with your team.");
      window.open("https://wa.me/" + phone + "?text=" + text, "_blank");
    }

    function openContactEmail() {
      // Change this to your main contact email
      window.location.href = "mailto:info@sabertechs.com?subject=Contact%20SaberTechs&body=Hi%20Team,%0D%0A%0D%0AWe%20would%20like%20to%20discuss%20a%20requirement.%20Here%20are%20some%20quick%20details:%0D%0A%0D%0A-%20Industry%3A%0D%0A-%20Type%20of%20work%20%28support%2Fsales%2Fback%20office%2FQA%2Fautomation%2Fgig%29%3A%0D%0A-%20Approx.%20volumes%20per%20day%2Fmonth%3A%0D%0A-%20Regions%20%2F%20time%20zones%3A%0D%0A%0D%0ARegards,%0D%0A";
    }

    // When form is submitted, open mailto with filled values
    function handleSubmit(event) {
      event.preventDefault();
      const form = event.target;

      const name = encodeURIComponent(form.name.value || "");
      const company = encodeURIComponent(form.company.value || "");
      const email = encodeURIComponent(form.email.value || "");
      const phone = encodeURIComponent(form.phone.value || "");
      const service = encodeURIComponent(form.service.value || "");
      const region = encodeURIComponent(form.region.value || "");
      const volumes = encodeURIComponent(form.volumes.value || "");
      const requirement = encodeURIComponent(form.requirement.value || "");
      const contactPref = encodeURIComponent(form.contact_preference.value || "");
      const time = encodeURIComponent(form.time.value || "");

      const subject = "Contact form submission – SaberTechs website";
      const body =
        "Hi Team,%0D%0A%0D%0AHere are the details from the contact form:%0D%0A%0D%0A" +
        "Name: " + name + "%0D%0A" +
        "Company: " + company + "%0D%0A" +
        "Email: " + email + "%0D%0A" +
        "Phone/WhatsApp: " + phone + "%0D%0A" +
        "Service interested in: " + service + "%0D%0A" +
        "Region/Market: " + region + "%0D%0A" +
        "Approx. volumes: " + volumes + "%0D%0A" +
        "Requirement details: " + requirement + "%0D%0A" +
        "Contact preference: " + contactPref + "%0D%0A" +
        "Preferred time to connect (IST): " + time + "%0D%0A%0D%0A" +
        "Regards,%0D%0A" + name;

      const mailto = "mailto:info@sabertechs.com"
        + "?subject=" + encodeURIComponent(subject)
        + "&body=" + body;

      window.location.href = mailto;
    }
  </script>
